pnl_lower = merged_data['Closed PnL'].quantile(0.01)
pnl_upper = merged_data['Closed PnL'].quantile(0.99)
clean_data = merged_data[(merged_data['Closed PnL'] >= pnl_lower) & (merged_data['Closed PnL'] <= pnl_upper)]

sentiment_performance = clean_data.groupby('classification').agg({
    'Closed PnL': ['count', 'mean', 'sum', 'std']
}).round(2)

correlation = clean_data['value'].corr(clean_data['Closed PnL'])
clean_data['is_profitable'] = clean_data['Closed PnL'] > 0
win_rate_by_sentiment = clean_data.groupby('classification')['is_profitable'].mean()

fig, axes = plt.subplots(2, 2, figsize=(15, 12))

sentiment_order = ['Extreme Fear', 'Fear', 'Neutral', 'Greed', 'Extreme Greed']
available_sentiments = [s for s in sentiment_order if s in clean_data['classification'].unique()]

# Plot 1: Average PnL by sentiment (clean)
avg_pnl = clean_data.groupby('classification')['Closed PnL'].mean().reindex(available_sentiments)
axes[0,0].bar(avg_pnl.index, avg_pnl.values, color=['red', 'orange', 'gray', 'lightgreen', 'green'])
axes[0,0].set_title('Average PnL by Market Sentiment (Outliers Removed)')
axes[0,0].tick_params(axis='x', rotation=45)

# Plot 2: Win rate by sentiment
axes[0,1].bar(win_rate_by_sentiment.index, win_rate_by_sentiment.values * 100, 
              color=['red', 'orange', 'gray', 'lightgreen', 'green'])
axes[0,1].set_title('Win Rate by Market Sentiment (%)')
axes[0,1].tick_params(axis='x', rotation=45)

# Plot 3: PnL distribution by sentiment (boxplot)
sns.boxplot(data=clean_data, x='classification', y='Closed PnL', order=available_sentiments, ax=axes[1,0])
axes[1,0].set_title('PnL Distribution by Sentiment')
axes[1,0].tick_params(axis='x', rotation=45)

# Plot 4: Clean scatter plot
axes[1,1].scatter(clean_data['value'], clean_data['Closed PnL'], alpha=0.6, s=10)
axes[1,1].set_xlabel('Fear & Greed Index')
axes[1,1].set_ylabel('Closed PnL')
axes[1,1].set_title(f'Fear & Greed vs PnL (Correlation: {correlation:.3f})')

plt.tight_layout()
plt.show()
